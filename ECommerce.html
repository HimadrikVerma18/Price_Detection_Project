import requests import pandas as pd from typing import Dict, List class ProductDataCollector: def __init__(self, api_key: str): self.api_key = api_key self.base_url = "https://api.ecommerce.com/v1" def fetch_product_data(self, category: str, limit: int = 1000) -> pd.DataFrame: """Fetch product data from API""" headers = {"Authorization": f"Bearer {self.api_key}"} response = requests.get( f"{self.base_url}/products", params={"category": category, "limit": limit}, headers=headers ) return pd.DataFrame(response.json()['products'])

def preprocess_data(df: pd.DataFrame) -> pd.DataFrame: # Handle missing values df['weight'] = df['weight'].fillna(df['weight'].median()) df['brand_rating'] = df['brand_rating'].fillna(3.0) # Feature engineering df['price_per_weight'] = df['price'] / df['weight'] df['volume'] = df['length'] * df['width'] * df['height'] df['is_premium'] = (df['brand_rating'] >= 4.5).astype(int) return df

from flask import Flask, request, jsonify import joblib import pandas as pd import numpy as np from datetime import datetime app = Flask(__name__) # Load trained model model = joblib.load('models/price_prediction_model.pkl') scaler = joblib.load('models/feature_scaler.pkl') @app.route('/predict', methods=['POST']) def predict_price(): """API endpoint for price prediction""" try: data = request.json # Prepare features features = pd.DataFrame([{ 'category': data['category'], 'brand_rating': data['brand_rating'], 'weight': data['weight'], 'dimensions': data['dimensions'], 'warranty': data['warranty'], 'seller_rating': data['seller_rating'], 'competition': data['competition'] }]) # Preprocess features features_scaled = scaler.transform(features) # Make prediction prediction = model.predict(features_scaled)[0] confidence = model.predict_proba(features_scaled)[0].max() # Anomaly detection is_anomaly, explanation = detect_anomaly(features, prediction) return jsonify({ 'predicted_price': round(prediction, 2), 'confidence': round(confidence, 3), 'is_anomaly': is_anomaly, 'explanation': explanation, 'timestamp': datetime.now().isoformat() }) except Exception as e: return jsonify({'error': str(e)}), 400 def detect_anomaly(features, prediction): """Detect price anomalies and generate explanations""" # Simple anomaly detection logic expected_range = get_price_range(features['category'].iloc[0]) if prediction < expected_range[0] * 0.5: return True, f"Price significantly below market average for {features['category'].iloc[0]} category" elif prediction > expected_range[1] * 2: return True, f"Price significantly above market average for {features['category'].iloc[0]} category" return False, "Price within expected range" if __name__ == '__main__': app.run(debug=True, host='0.0.0.0', port=5000)
